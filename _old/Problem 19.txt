Module Module1

    Sub Main()

        Dim endTime As Double
        Dim starTime As Double = Timer
        Dim resposta As Integer = 0

        Dim dia As Integer = 1
        Dim mes As Integer = 1
        Dim ano As Integer = 1900
        Dim weekDay As Integer = 1

        Do While Not (dia = 31 And mes = 12 And ano = 2000)


            dia = GenerateNewDay(dia, mes, ano)
            If dia = 1 And mes = 12 Then
                mes = 1
            ElseIf dia = 1 Then
                mes += 1
            End If
            If dia = 1 And mes = 1 Then
                ano += 1
            End If

            If weekDay = 7 Then
                weekDay = 1
            Else
                weekDay += 1
            End If

            If dia = 1 And weekDay = 7 And ano > 1900 Then
                resposta += 1
            End If

        Loop


        endTime = Timer
        Console.WriteLine("Resposta: " & resposta)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()
    End Sub

    Function GenerateNewDay(dia As Integer, mes As Integer, ano As Integer) As Integer
        Dim newDay As Integer = dia
        Dim isLeapYear As Boolean

        If ano Mod 400 = 0 Then
            isLeapYear = True
        ElseIf ano Mod 100 = 0 Then
            isLeapYear = False
        ElseIf ano Mod 4 = 0 Then
            isLeapYear = True
        Else
            isLeapYear = False
        End If

        Select Case mes
            Case 2
                If dia = 29 Then
                    newDay = 1
                ElseIf dia = 28 And isLeapYear = False Then
                    newDay = 1
                Else
                    newDay += 1
                End If
            Case 4, 6, 9, 11
                If dia = 30 Then
                    newDay = 1
                Else
                    newDay += 1
                End If
            Case Else
                If dia = 31 Then
                    newDay = 1
                Else
                    newDay += 1
                End If
        End Select

        Return newDay
    End Function

End Module








using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Diagnostics;


namespace ConsoleApplication3
{
    class Program
    {
        static void Main(string[] args){

            int answer = 0;
            Stopwatch timer = new Stopwatch();
            timer.Start();

            int day = 1;
            int month = 1;
            int year = 1;
            int weekDay = 1;

            while (!(day == 31 && month == 12 && year == 2000)){

                day = GenerateNewDay(day, month, year);

                if (day == 1){
                    month = month == 12 ? 1 : month + 1;
                }
                
                if (day == 1 && month == 1){
                    year++;
                }
                
                weekDay = weekDay == 7 ? 1 : weekDay + 1;

                if (day == 1 && weekDay == 7 && year > 1900){
                    answer += 1;
                }
            }

            timer.Stop();
            Console.WriteLine("Answer: {0}", answer);
            Console.WriteLine("Processing time: {0}", timer.Elapsed);
            Console.ReadKey();
        }

        public static int GenerateNewDay(int day, int month, int year){

            int newDay = day;
            bool isLeapYear;

            if (year % 400 == 0){
                isLeapYear = true;
            } else if (year % 100 == 0){
                isLeapYear = false;
            } else if (year % 4 == 0){
                isLeapYear = true;
            } else {
                isLeapYear = false;
            }

            switch (month){
                case 2:
                    if(day == 29){
                        newDay = 1;
                    }else if(!isLeapYear && day == 28){
                        newDay = 1;
                    }else {
                        newDay++;
                    }
                    break;
                case 4: case 6: case 9: case 11:
                    newDay = day == 30? 1: newDay + 1;
                    break;
                default:
                    newDay = day == 31? 1: newDay + 1;
                    break;
            }

            return newDay;
        }
    }
}
