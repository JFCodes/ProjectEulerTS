Module Module1

    Sub Main()
        Dim endTime As Double
        Dim starTime As Double = Timer
        Dim resposta As String = ""

        Dim sequence As String = "0123456789"
        Dim iterations As Integer = 999999
        Dim counter As Short = 9
        Dim factorial As Integer
        Dim a As Short
        Dim digit As String

        Do While counter <> 0
            factorial = 1
            For i = 1 To counter
                factorial *= i
            Next

            a = 0
            Do While a * factorial <= iterations
                a += 1
            Loop
            a -= 1

            digit = sequence(a).ToString
            resposta &= digit
            sequence = Replace(sequence, digit, "")
            iterations = iterations - factorial * a

            counter -= 1
        Loop

        resposta &= sequence

        endTime = Timer
        Console.WriteLine("Resposta: " & resposta)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()
    End Sub

End Module













using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Diagnostics;





namespace ConsoleApplication3
{
    class Program
    {
        static void Main(string[] args)
        {

            string answer = "";
            Stopwatch timer = new Stopwatch();
            timer.Start();

            string sequence = "0123456789";
            int iterations = 999999;
            short counter = 9;
            int factorial;
            short a;
            char digit;

            while (counter != 0)
            {
                factorial = 1;
                for (int i = 1; i <= counter; i++)
                {
                    factorial *= i;
                }

                a = 0;
                while (a * factorial <= iterations)
                {
                    a++;
                }
                a--;

                digit = sequence[a];
                answer += Convert.ToString(digit);
                sequence = sequence.Remove(a, 1);
                iterations -= factorial * a;

                counter--;
            }

            answer += sequence;

            timer.Stop();
            Console.WriteLine("Answer: {0}", answer);
            Console.WriteLine("Processing time: {0}", timer.Elapsed);
            Console.ReadKey();
        }
    }
}
