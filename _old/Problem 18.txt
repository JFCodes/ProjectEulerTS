    Sub Main()
        Dim endTime As Double
        Dim resposta As Integer = -1

        Dim reader As New System.IO.StreamReader("C:\Users\Joaquin\Dropbox\Project Euler\Sources\Problem 18.txt")
        Dim line As String
        Dim splitLine() As String
        Dim lines As Integer = 0
        Dim totalLines As Integer = 0

        While reader.Peek <> -1
            reader.ReadLine()
            totalLines = totalLines + 1
        End While

        reader.Close()
        reader = New System.IO.StreamReader("C:\Users\Joaquin\Dropbox\Project Euler\Sources\Problem 18.txt")

        Dim triangle(totalLines, totalLines) As Integer

        Do While reader.Peek <> -1
            line = reader.ReadLine
            splitLine = Split(line, " ")

            For j = 0 To lines
                triangle(lines, j) = splitLine(j)
            Next
            For j = lines + 1 To totalLines
                triangle(lines, j) = 0
            Next
            lines += 1
        Loop

        reader.Dispose()
        Dim starTime As Double = Timer

        For i = 1 To totalLines
            For j = 0 To totalLines
                If j = 0 Then
                    triangle(i, j) = triangle(i, j) + triangle(i - 1, j)
                Else
                    triangle(i, j) = triangle(i, j) + Math.Max(triangle(i - 1, j), triangle(i - 1, j - 1))
                End If
            Next
        Next

        For j = 0 To totalLines
            If triangle(totalLines, j) > resposta Then resposta = triangle(totalLines, j)
        Next

        endTime = Timer
        Console.WriteLine("Resposta: " & resposta)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()
    End Sub

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Diagnostics;


namespace ConsoleApplication3
{
    class Program
    {
        static void Main(string[] args){

            int answer = -1;
            Stopwatch timer = new Stopwatch();
            timer.Start();

            string[] lines = System.IO.File.ReadAllLines(@"C:\Users\Joaquin\Dropbox\Project Euler\Sources\Problem 18.txt");       
            int[,] triangle = new int[lines.Length, lines.Length];

            for (int i = 0; i < lines.Length; i++){
                string[] splitline = lines[i].Split(' ');

                for (int j = 0; j < lines.Length; j++)
                {
                    triangle[i, j] = j >= splitline.Length ? 0 : Convert.ToInt32(splitline[j]);
                }
            }

            for (int i = 1; i < lines.Length; i++) {
                for (int j = 0; j < lines.Length; j++){
                    if (j == 0) {
                        triangle[i, j] = triangle[i, j] + triangle[i - 1, j];
                    } else {
                        triangle[i, j] = triangle[i, j] + Math.Max(triangle[i - 1, j], triangle[i - 1, j - 1]);
                    }
                }
            }

            for (int j = 0; j < lines.Length; j++){
                answer = triangle[lines.Length - 1, j] > answer ? triangle[lines.Length- 1, j] : answer;
            }


            timer.Stop();
            Console.WriteLine("Answer: {0}", answer);
            Console.WriteLine("Processing time: {0}", timer.Elapsed);
            Console.ReadKey();
        }
    }
}
