Module Module1

    Sub Main()

        Dim endTime As Double
        Dim starTime As Double = Timer
        Dim answer As Long = 0

        Dim nextN As Integer = 2
        Dim stringCheck As String = 0

        For i = 9999 To 1 Step -1
            stringCheck = CStr(i)
            nextN = 1

            Do While stringCheck.Length <= 9
                nextN += 1
                stringCheck &= CStr(i * nextN)

                If stringCheck.Length = 9 _
                    And InStr(stringCheck, "1") > 0 _
                    And InStr(stringCheck, "2") > 0 _
                    And InStr(stringCheck, "3") > 0 _
                    And InStr(stringCheck, "4") > 0 _
                    And InStr(stringCheck, "5") > 0 _
                    And InStr(stringCheck, "6") > 0 _
                    And InStr(stringCheck, "7") > 0 _
                    And InStr(stringCheck, "8") > 0 _
                    And InStr(stringCheck, "9") > 0 Then

                    answer = CInt(stringCheck)
                End If
            Loop
        Next

        endTime = Timer
        Console.WriteLine("Answer: " & answer)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()

    End Sub

End Module







using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Diagnostics;


namespace ConsoleApplication3
{
    class Program
    {
        static void Main(string[] args)
        {

            int answer = 0;
            Stopwatch timer = new Stopwatch();
            timer.Start();

            long nextN;
            string stringCheck;

            for (int i = 9999; i > 0; i--){
                stringCheck = Convert.ToString(i);
                nextN = 1;

                while (stringCheck.Length <= 9){
                    nextN++;
                    stringCheck += Convert.ToString(i * nextN);

                    if( stringCheck.Length == 9 && stringCheck.Contains('1')
                        && stringCheck.Contains('2') && stringCheck.Contains('3')
                        && stringCheck.Contains('4') && stringCheck.Contains('5') 
                        && stringCheck.Contains('6') && stringCheck.Contains('7')
                        && stringCheck.Contains('8') && stringCheck.Contains('9')){

                            answer = Convert.ToInt32(stringCheck) > answer ? Convert.ToInt32(stringCheck) : answer;
                    }
                    
                }
            }


            timer.Stop();
            Console.WriteLine("Answer: {0}", answer);
            Console.WriteLine("Processing time: {0}", timer.Elapsed);
            Console.ReadKey();
        }
    }
}
