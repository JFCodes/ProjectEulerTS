Module Module1

    Sub Main()
        Dim endTime As Double
        Dim starTime As Double = Timer
        Dim resposta As Integer = 0

        Dim runnerUp As Long = 0
        Dim increment As Integer = 1
        Dim strinSrip As String = ""
        Dim valores(0) As Integer
        Dim position As Integer = 0
        Dim firsValue As Boolean = True
        Dim foundValue As Boolean = False

        For i = 1 To 9999

            increment = 1
            runnerUp = i * increment
            strinSrip = CStr(i) & CStr(increment) & CStr(runnerUp)

            Do While strinSrip.Length <= 9

                If strinSrip.Length = 9 And _
                    InStr(strinSrip, "1") > 0 And _
                    InStr(strinSrip, "2") > 0 And _
                    InStr(strinSrip, "3") > 0 And _
                    InStr(strinSrip, "4") > 0 And _
                    InStr(strinSrip, "5") > 0 And _
                    InStr(strinSrip, "6") > 0 And _
                    InStr(strinSrip, "7") > 0 And _
                    InStr(strinSrip, "8") > 0 And _
                    InStr(strinSrip, "9") > 0 Then

                    If firsValue Then
                        valores(0) = runnerUp
                        resposta = runnerUp
                        firsValue = False
                    Else

                        foundValue = False
                        For j = 0 To valores.Length - 1
                            If valores(j) = runnerUp Then
                                foundValue = True
                            End If
                        Next
                        If foundValue = False Then
                            position += 1
                            ReDim Preserve valores(position)
                            valores(position) = runnerUp
                            resposta += runnerUp
                        End If

                    End If
                End If

                increment += 1
                runnerUp = i * increment
                strinSrip = CStr(i) & CStr(increment) & CStr(runnerUp)
            Loop

        Next

        endTime = Timer
        Console.WriteLine("Resposta: " & resposta)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()
    End Sub


End Module











using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Diagnostics;


namespace ConsoleApplication3
{
    class Program
    {
        static void Main(string[] args){

            int answer = 0;
            Stopwatch timer = new Stopwatch();
            timer.Start();

            int runnerUp = 0;
            int increment = 1;
            bool firstValue = true, foundValue = false;
            string stringStrip = "";
            List<int> values = new List<int>();

            for (int i = 1; i <= 999; i++){

                increment = 1;
                runnerUp = i * increment;
                stringStrip = Convert.ToString(i) + Convert.ToString(increment) + Convert.ToString(runnerUp);

                while (stringStrip.Length <= 9){
                
                    if(stringStrip.Length == 9 && 
                        stringStrip.Contains('1') &&  
                        stringStrip.Contains('2') &&  
                        stringStrip.Contains('3') &&  
                        stringStrip.Contains('4') &&  
                        stringStrip.Contains('5') &&  
                        stringStrip.Contains('6') &&  
                        stringStrip.Contains('7') &&  
                        stringStrip.Contains('8') &&
                        stringStrip.Contains('9')){
   
                        if (!values.Contains(runnerUp)){
                            values.Add(runnerUp);
                            answer += runnerUp;
                        }
                    }

                    increment++;
                    runnerUp = i * increment;
                    stringStrip = Convert.ToString(i) + Convert.ToString(increment) + Convert.ToString(runnerUp);
                }
            }

                timer.Stop();
            Console.WriteLine("Answer: {0}", answer);
            Console.WriteLine("Processing time: {0}", timer.Elapsed);
            Console.ReadKey();
        }
    }
