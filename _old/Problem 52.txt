Module Module1

    Sub Main()
        Dim endTime As Double
        Dim starTime As Double = Timer
        Dim resposta As Integer

        Dim increment As Integer = 123456

        Do
            If CStr(increment).Length = CStr(6 * increment).Length Then
                If checkNumbers(2 * increment, 3 * increment, 4 * increment, 5 * increment, 6 * increment) Then
                    resposta = increment
                    Exit Do
                End If
            End If
            increment += 1
        Loop

        endTime = Timer
        Console.WriteLine("Resposta: " & resposta)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()
    End Sub

    Function checkNumbers(n2 As Integer, n3 As Integer, n4 As Integer, n5 As Integer, n6 As Integer) As Boolean

        Dim n2vector() As Char = n2.ToString.ToCharArray
        Dim n3vector() As Char = n3.ToString.ToCharArray
        Dim n4vector() As Char = n4.ToString.ToCharArray
        Dim n5vector() As Char = n5.ToString.ToCharArray
        Dim n6vector() As Char = n6.ToString.ToCharArray

        Array.Sort(n2vector)
        Array.Sort(n3vector)
        Array.Sort(n4vector)
        Array.Sort(n5vector)
        Array.Sort(n6vector)

        If n2vector <> n3vector Or _
            n3vector <> n4vector Or _
            n4vector <> n5vector Or _
            n5vector <> n6vector Then
            Return False
        End If

        Return True
    End Function

End Module
