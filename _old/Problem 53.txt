Module Module1

    Sub Main()
        Dim endTime As Double
        Dim starTime As Double = Timer
        Dim resposta As Integer = 0

        Dim combinations As Double

        For n = 1 To 100
            For r = 1 To n

                combinations = (factorial(n)) / (factorial(r) * factorial(n - r))
                If combinations > 1000000 Then resposta += 1
            Next
        Next

        endTime = Timer
        Console.WriteLine("Resposta: " & resposta)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()
    End Sub

    Function factorial(number As Integer) As Double
        factorial = 1

        For i = 1 To number
            factorial *= i
        Next
    End Function
End Module
