using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Diagnostics;


namespace ConsoleApplication3
{
    class Program
    {
        static void Main(string[] args)
        {

            int answer = 0;
            Stopwatch timer = new Stopwatch();

            string text = System.IO.File.ReadAllText(@"C:\Users\Joaquin\Dropbox\Project Euler\Sources\Problem 42.txt");
            string[] names = text.Replace("\"","").Split(',');
            int wordScore;

            timer.Start();
            foreach(string name in names){
                wordScore = 0;

                foreach (char letter in name){
                    wordScore += (int)letter - 64;
                }

                answer += (-1 + Math.Sqrt(1 + (8 * wordScore))) / 2 % 1 == 0 ? 1 : 0;
            }

            timer.Stop();
            Console.WriteLine("Answer: {0}", answer);
            Console.WriteLine("Processing time: {0}", timer.Elapsed);
            Console.ReadKey();
        }
    }
}
