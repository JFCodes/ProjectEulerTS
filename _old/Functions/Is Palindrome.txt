Function isPalindrome(number As Double) As Boolean

        Dim numbers() As Char = number.ToString.ToCharArray

        For i = 0 To ((numbers.Length - 1) \ 2)
            If numbers(i) <> numbers(numbers.Length - i - 1) Then Return False
        Next

        Return True
    End Function
