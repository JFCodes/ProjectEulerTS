Module Module1

    Sub Main()

        Dim endTime As Double
        Dim starTime As Double = Timer
        Dim answer As Long = 0

        Dim numberString As String = ""
        Dim numberBinary As String = ""

        For i = 1 To 1000000 Step 2

            numberString = CStr(i)

            If numberString = StrReverse(numberString) Then

                numberBinary = Convert.ToString(i, 2)
                If numberBinary = StrReverse(numberBinary) Then
                    answer += i
                End If
            End If

        Next

        endTime = Timer
        Console.WriteLine("Answer: " & answer)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()

    End Sub

End Module











using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Diagnostics;


namespace ConsoleApplication3
{
    class Program
    {
        static void Main(string[] args)
        {

            int answer = 0;
            Stopwatch timer = new Stopwatch();
            timer.Start();

            string numberString, binaryString;
            string numberStringReverse, binaryStringReverse;

            for (int i = 1; i < 1000000; i += 2){
                numberString = Convert.ToString(i);
                
                char[] charArray = numberString.ToCharArray();
                Array.Reverse(charArray);
                numberStringReverse = new string(charArray);

                if (numberString == numberStringReverse) {

                    binaryString = Convert.ToString(i, 2);
                    charArray = binaryString.ToCharArray();
                    Array.Reverse(charArray);
                    binaryStringReverse = new string(charArray);
                   
                    if(binaryString == binaryStringReverse) {
                        answer += i;
                    }
                    
                }
            }
            
            timer.Stop();
            Console.WriteLine("Answer: {0}", answer);
            Console.WriteLine("Processing time: {0}", timer.Elapsed);
            Console.ReadKey();
        }
    }
}
