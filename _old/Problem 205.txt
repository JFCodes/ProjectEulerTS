Module Module1

    Sub Main()
        Dim endTime As Double
        Dim starTime As Double = Timer
        Dim resposta As Double = 0


        Dim vectorNines(36) As Integer
        Dim vectorSixes(36) As Integer
        Dim sumNines As Integer = 0
        Dim sumSixes As Integer = 0

        For a = 1 To 4
            For b = 1 To 4
                For c = 1 To 4
                    For d = 1 To 4
                        For e = 1 To 4
                            For f = 1 To 4
                                For g = 1 To 4
                                    For h = 1 To 4
                                        For j = 1 To 4
                                            vectorNines(a + b + c + d + e + f + g + h + j) += 1
                                            sumNines += 1
                                        Next
                                    Next
                                Next
                            Next
                        Next
                    Next
                Next
            Next
        Next
        For a = 1 To 6
            For b = 1 To 6
                For c = 1 To 6
                    For d = 1 To 6
                        For e = 1 To 6
                            For f = 1 To 6
                                vectorSixes(a + b + c + d + e + f) += 1
                                sumSixes += 1
                            Next
                        Next
                    Next
                Next
            Next
        Next

        For i = 9 To 36
            Dim favoraveisNine = vectorNines(i)
            Dim favoraveisSix = 0
            For j = 6 To i - 1
                favoraveisSix += vectorSixes(j)
            Next
            resposta += (favoraveisNine / sumNines) * (favoraveisSix / sumSixes)
        Next

        endTime = Timer
        Console.WriteLine("Resposta: " & resposta)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()
    End Sub


End Module
