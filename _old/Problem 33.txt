Module Module1

    Sub Main()
        Dim endTime As Double
        Dim starTime As Double = Timer

        Dim denproduct As Integer = 1
        Dim nomproduct As Integer = 1

        For i = 1 To 9
            For den = 1 To i - 1
                For nom = 1 To den - 1
                    If ((nom * 10) + i) * den = nom * ((i * 10) + den) Then
                        denproduct *= den
                        nomproduct *= nom
                    End If
                Next
            Next
        Next

        endTime = Timer
        Console.WriteLine("Answer: " & nomproduct & " / " & denproduct)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()

    End Sub

End Module






using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Diagnostics;


namespace ConsoleApplication3
{
    class Program
    {
        static void Main(string[] args){

            Stopwatch timer = new Stopwatch();
            timer.Start();

            int denProduct = 1, nomProduct = 1;

            for (int i = 1; i <= 9; i++){
                for (int den = 1; den <= i - 1; den++){
                    for (int nom = 1; nom <= den - 1; nom++){
                        if (((nom * 10) + i) * den == nom * ((i * 10) + den)){
                            denProduct *= den;
                            nomProduct *= nom;
                        }
                    }
                }
            }
                
            timer.Stop();
            Console.WriteLine("Answer: {0} / {1}", denProduct, nomProduct);
            Console.WriteLine("Processing time: {0}", timer.Elapsed);
            Console.ReadKey();
        }
    }
}
