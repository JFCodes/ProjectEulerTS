Module Module1

    Sub Main()
        Dim endTime As Double
        Dim starTime As Double = Timer
        Dim resposta As Integer = 0

        Dim sumDivisors1 As Integer
        Dim sumDivisors2 As Integer

        For i = 2 To 10000

            sumDivisors1 = sumOfDivisors(i)
            sumDivisors2 = sumOfDivisors(sumDivisors1)

            If sumDivisors2 = i And i <> sumDivisors1 Then
                resposta += i
            End If
        Next

        endTime = Timer
        Console.WriteLine("Resposta: " & resposta)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()
    End Sub

    Function sumOfDivisors(number As Integer)
        sumOfDivisors = 1

        For i = 2 To Math.Sqrt(number)
            If number Mod i = 0 Then
                sumOfDivisors += i
                sumOfDivisors += number / i
            End If
        Next

    End Function

End Module



using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Diagnostics;


namespace ConsoleApplication3
{
    class Program
    {
        static void Main(string[] args){

            int answer = 0;
            Stopwatch timer = new Stopwatch();
            timer.Start();

            int sumDivisor1, sumDivisor2;

            for (int i = 2; i <= 10000; i++){
                sumDivisor1 = sumOfDivisors(i);
                sumDivisor2 = sumOfDivisors(sumDivisor1);

                if (sumDivisor2 == i && i != sumDivisor1){
                    answer += i;
                }
            }

            timer.Stop();
            Console.WriteLine("Answer: {0}", answer);
            Console.WriteLine("Processing time: {0}", timer.Elapsed);
            Console.ReadKey();
        }

        public static int sumOfDivisors(int number){
            int sum = 1;

            for (int i = 2; i <= Math.Sqrt(number); i++){
                if (number % i == 0){
                    sum += i + number / i;
                }
            }

            return sum;
        }
    }
}
