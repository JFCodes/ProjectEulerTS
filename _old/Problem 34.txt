Module Module1

    Sub Main()
        Dim endTime As Double
        Dim starTime As Double = Timer
        Dim resposta As Integer = 0

        Dim factorials(9) As Long
        Dim sumOfFactors As Long = 0
        Dim deno As Integer

        factorials(0) = 1
        For i = 1 To 9
            factorials(i) = factorials(i - 1) * i
        Next


        For i = 3 To 2540161

            deno = i
            sumOfFactors = 0
            Do While deno <> 0
                sumOfFactors += factorials(deno Mod 10)
                deno \= 10
            Loop

            If sumOfFactors = i Then
                resposta += i
            End If
        Next

        endTime = Timer
        Console.WriteLine("Resposta: " & resposta)
        Console.WriteLine("Excution time: " & endTime - starTime)
        Console.ReadKey()
    End Sub


End Module




using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Diagnostics;


namespace ConsoleApplication3
{
    class Program
    {
        static void Main(string[] args){

            int answer = 0;
            Stopwatch timer = new Stopwatch();
            timer.Start();

            long[] factorials = new long[10];
            long sumFactors = 0;
            int deno;

            factorials[0] = 1;
            for (int i = 1; i <= 9; i++){
                factorials[i] = i * factorials[i - 1];
            }

            for (int i = 3; i <= 2540161; i++){
                deno = i;
                sumFactors = 0;
                while (deno != 0){
                    sumFactors += factorials[deno % 10];
                    deno /= 10;
                }

                answer += sumFactors == i ? i : 0;
            }
            
            
            timer.Stop();
            Console.WriteLine("Answer: {0}", answer);
            Console.WriteLine("Processing time: {0}", timer.Elapsed);
            Console.ReadKey();
        }
    }
}
